ES6语法

无标签
发布日期:   2020-03-31
文章字数:   940

let

ES6中新增的用于声明变量的关键字

  • let声明的变量只在所处于的块级有效

    if(true){
    let a = 10;
    var b = 10;
}
console.log(a); //a is not define
console.log(b); // 打印输出10

  • 不存在变量提升

    console.log(a); // a is not define
let a = 20; // 必须先声明，再使用

  • 暂时性死区

    // 在块级内不会受到外部影响
var tmp = 123;
if(true){
    console.log(tmp); // tmp is not define;
    let tmp; // 会和整体进行绑定
}

const

声明常量

  • 具有块级作用域

    if(true){
    const a = 10;
}
console.log(a); // a is not define

  • 声明常量时必须赋值

    const PI; // Missing initializer in const declaration

  • 常量赋值后，不能更改

    const PI = 3.14;
PI =  100; // Assignment to constant variable

const ary = [10,20];
ary[0] = 'a'; // 不会报错
ary = ['a','b']; // Assignment to constant variable

解构赋值

允许从数组中提取值，按照对应位置，对变量赋值。对象也可以实现解构

  • 数组解构

    let [a,b,c] = [1,2,3]; // 允许按照一一对应的关系从数组中提取值，再将值赋值给变量
console.log(a); // 1
console.log(b); // 2

  • 对象解构

    let person = {name: '张三', age: 10};
let {name,age} = person;
console.log(name); // 张三
//另一种写法
let {name:myName,age:myAge} = person; //myName属于别名
console.log(myName); // 张三

箭头函数

() => {}
const fn = () => {} 

//如果函数只有一句话，且代码的执行结果就是返回值，可以省略大括号
const sum = (num1,num2) => num1 + num2;

// 如果函数的形参只有一个时，小括号可以省略
const fn = v => v;

// 箭头函数不绑定this,箭头函数中的this,指向的是函数定义位置的上下文this

剩余参数

剩余参数允许我们将一个不定数量的参数表示为一个数组

function sum (first, ...args){
    console.log(args); // [20,30]
}
sum(10,20,30)
// 剩余参数和解构配合使用
let students = ['wangwu','zhangsan','lisi'];
let [s1,...s2] = students;
console.log(s1); // 'wangwu'
console.log(s2); // ['zhangsan','lisi']

扩展运算符

将数组或对象拆分成逗号分隔的参数序列

 let ary = [1,2,3];
...ary
console.log(...ary); // 1 2 3

//用来合并数组
let ary1 = [1,2,3];
let ary2 = [3,4,5];
let ary3 = [...ary1,...ary2];
ary1.push(...ary2);

//将伪数组转换为真正的数组
let Divs = document.getElementsByTagName('div');
Divs = [...Divs];

Array的扩展方法

find()方法

找到第一个符合条件的数组成员

let ary = [1,2,3,4,5];
let target = ary.find((item) => item == 2);

findIndex()方法

找到第一个符合条件的数组成员的位置

let ary = [1,2,3,4,5];
let index = ary.find((item) => item == 2);

includes()方法

表示某个数组是否包含给定的值，返回布尔值

[1,2,3].includes(2); //true

String的扩展方法

模板字符串

let name = 'zhangsan'
// 模板字符串可以解析变量
let hello = `hello,myname is $name`; // hello,myname is zhangsan

// 模板字符串可以换行
let html = `hello,myname 
is $name`; // hello,myname is zhangsan

//模板字符串可以调用函数
const sayhello = function(){
    return "hello";
}
let greet = `${sayhello()}`; //字符串内容是返回的内容
console.log(greet); // hello

startsWith()和endsWith()

分别来判断字符串是否是参数开头和结尾

let str = 'Hello World';
str.startsWith('Hello') // true

repeat()

表示将字符串重复n次，返回一个新的字符串

let str1 = 'a';
let str2 = str1.repeat(3); // aaa

Set数据结构

集合数据结构

const s = new Set([1,2,3,4,4]); // 1,2,3,4

add(value)

添加某个值，返回set结构本身

delete(value)

删除某个值，返回一个布尔值

has(value)

返回一个布尔值，判断value是否在里面

clear()

清除所有成员，没有返回值

遍历

s.forEach(value => console.log(value))
